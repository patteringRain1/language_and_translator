coll Card {
    STRING name;
    INT damage;
    BOOL isMagic;
}

coll Player {
    INT hp;
    Card[] deck;
}

def INT calculateDamage(Card c, BOOL enemyResistsMagic) {
    if (c.isMagic && enemyResistsMagic) {
        return c.damage / 2;
    }
    return max(0, c.damage);
}
